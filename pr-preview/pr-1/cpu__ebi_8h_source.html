    45  /* place 24 bit address into r30, r31, RAMPZ
    46  * r31:r30 <- %B[from]:%A[from]
    47  * RAMPZ <- %C[from]
    48  */
    49  "movw r30, %A[from] \n\t"
    50  "out %[rampz], %C[from] \n\t"
