    808 {

    814 #ifdef SOCK_HAS_IPV4
    815  case AF_INET:
    816  return ipv4_addr_is_multicast((const ipv4_addr_t *)&ep->addr.ipv4);
    817 #endif
    818  default:
    819  assert(0);
