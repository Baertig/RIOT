    160 static inline gpio_port_t gpio_get_port(gpio_t pin)
    161 {
    162  /* GPIO LL and legacy GPIO API may disagree on what is the GPIO base
    163  * address if one is using the IOBUS and the other is using the APB for
    164  * access. In this case, we need to do impedance matching by adding the
    165  * offset. */
    166  const uintptr_t gpio_ll_base = GPIO_PORT_0;
    167  const uintptr_t gpio_legacy_base = GPIO_PIN(0, 0) & ~(0x1f);
    168  uintptr_t addr = (pin & ~(0x1f));
    169 
    170  return addr + (gpio_ll_base - gpio_legacy_base);
    171 }
