    988 static inline ssize_t gcoap_request(coap_pkt_t *pdu, uint8_t *buf, size_t len,
    989  unsigned code, char *path)
    990 {
    991  if (gcoap_req_init(pdu, buf, len, code, path) == 0) {
    992  if (IS_USED(MODULE_NANOCOAP_CACHE)) {
    993  /* remove ETag option slack added for cache validation */
    994  coap_opt_remove(pdu, COAP_OPT_ETAG);
    995  }
    996  return coap_opt_finish(pdu, COAP_OPT_FINISH_NONE);
    997  }
    998  return -1;
    999 }
