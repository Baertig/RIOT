    470 static inline bool ipv6_addr_is_global(const ipv6_addr_t *addr)
    471 {
    472  /* first check for multicast with global scope */
    473  if (ipv6_addr_is_multicast(addr)) {
    474  return ((addr->u8[1] & 0x0f) == IPV6_ADDR_MCAST_SCP_GLOBAL);
